plaintext
│──.husky/                        # husky配置
│──.vscode/                       # 私有化配置文件夹
│── build/                        # 构建打包文件夹
|── docs/                         # 文档说明文件夹
│
├── public/                       # 公共资源文件夹
├── src/                          # 源代码文件夹
│ ├── api/                        # 接口文件地址
│ ├── assets/                     # 静态资源文件夹 (图片、样式、svg文件、字体等)
│ ├── components/                 # 组件文件夹
│ ├── directive/                  # 全局指令封装
│ ├── hooks/                      # 自定义hooks封装存放地址
│ ├── layout/                     # 公共布局页面 首页
│ ├── views/                      # 页面以及组件
│ ├── router/                     # 路由文件夹，存放路由配置
│ ├── store/                      # 全局状态管理文件夹
│ ├── utils/                      # 工具类文件夹，存放工具函数
│ │── views/                      # 业务文件代码
│ │────────|custom_module         # 业务模块
│ │────────│   ├── api            # 业务模块私有接口
│ │────────│   ├── components     # 业务组件(涉及业务处理)
│ │────────│   ├── composable     # 业务组件(不涉及具体业务)
│ │────────│   ├── functional     # 业务函数式组件
│ │────────│   ├── methods/hooks  # 业务hooks
│ │────────│   ├── config         # 业务配置项
│ │────────│   ├── styles         # 业务样式
│ │────────│   └── utils          # 业务私有工具类
│ │────────|── index.vue          # 业务入口文件
│ │────────└── README.md          # 业务说明文档
│ ├── App.vue                     # 根组件
│ └── main.js                     # 项目入口文件
│ └── permission.js               # 全局路由拦截器
|
|── tests/                        # 测试用例文件夹
├── types/                        # 类型限制文件夹
|
|── .env.development              # 开发环境配置
|── .env.production               # 生产环境配置
|── .env.staging                  # 预发布环境配置
|── .eslintignore                 # eslint忽略文件配置
|── .eslintrc-auto-import.json    #eslint自动导入配置
|── .eslintrc.cjs                 # eslint配置
|── .gitattributes                # git属性配置
|── .gitignore                    # git忽略文件配置
|── .npmrc                        # npm配置
|── .nvmrc                        # nvm配置
|── .prettierignore               # prettier忽略文件配置
|── .prettierrc                   # prettier配置
|── CHANGELOG.md                  # 更新日志
|── CODECONVENTION.md             # 编码规范文档
|── commitlint.config.js          # commitlint配置
|── cspell.json                   # cspell配置
├── index.html                    # 单页主结构页面
├── LICENSE                       # 项目许可证
├── package.json                  # 项目配置文件
├── README.md                     # 说明文档
|── stylelint.config.js           # stylelint配置
|── tsconfig.json                 # ts配置文件
